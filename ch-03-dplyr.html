contains

function (match, ignore.case = TRUE, vars = current_vars()) 
{
    stopifnot(is.string(match), nchar(match) > 0)
    if (ignore.case) {
        vars <- tolower(vars)
        match <- tolower(match)
    }
    grep_vars(match, vars, fixed = TRUE)
}
